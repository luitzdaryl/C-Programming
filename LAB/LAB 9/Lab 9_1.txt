#include<stdio.h>

int main(){
    int firstSet[5], secondSet[5], printResult[5];
    int i, j, k=0;

      for(i=0;i<5;i++)
         scanf("%d",&firstSet[i]);  // taking input of set A

      for(i=0;i<5;i++)
         scanf("%d",&secondSet[i]); // taking input set B

    // Logic for intersection
      for( i=0;i<5;i++){
         for(j=0;j<5;j++){
            if(firstSet[i]==secondSet[j]){
                printResult[k]=firstSet[i];
                k++;
            }
         }
    }
      for(i=0;i<k;i++)
          printf("%d ",printResult[i]); // Printing the elements of intersection of set A and set B

    return 0;
}

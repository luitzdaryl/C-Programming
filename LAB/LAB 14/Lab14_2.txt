#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

int main(){
	FILE *ptr=NULL;
	FILE *ptr2=NULL;
	FILE *ptr3=NULL;

	int n1, n2;
	float f1, f2;

	int counter=0, c=0;
	bool checker=false;


	ptr=fopen("14_2trans.txt", "r");
	ptr2=fopen("14_2oldmast.txt", "r");
	ptr3=fopen("14_2newmast.txt", "w");
	char c1[50]={' '}, c2[50]={' '};
	
	while(c<5){
		fscanf(ptr, "%d %f", &n1, &f1);
		counter=0;
		checker=false;
		
		while(counter<3){
			fscanf(ptr2, "%d %s %s %f", &n2, &c1, &c2, &f2);

			if(n1==n2) {
				printf("%d %s %s %.2f\n", n1, c1, c2, f1+f2);

			checker=true;
			fprintf(ptr3, "%d %s %s %.2f\n", n1, c1, c2, f1+f2);
				break;
			}
			counter++;
		}
      
		if(checker==false){
			printf("Unmatched transaction record for account number %d\n", n1);
		}
		c++;
      
		//close the files
		fclose(ptr2);
		ptr2=NULL;
		ptr2=fopen("14_2oldmast.txt", "r");
	}
  
	fclose(ptr3);
	return 0;
}

#include <string.h>

#define SIZE 200

int alpha(char alphabet);// function prototype for the alphabet
int testPalindrome(char str[],int i,int j);//function prototype to test whether the string is palindrome or not

int main(){

char userInput[SIZE];
  
fgets(userInput,SIZE,stdin);

for(int i=0;i<strlen(userInput);i++){
      userInput[i]=tolower(userInput[i]);
  }
  
     if(testPalindrome(userInput,0,strlen(userInput)-1)== 1){
           printf("True");
}
     else{
           printf("False");
  }
  
return 0;
}

int alpha(char alphabet){
    
   if(alphabet>='a' && alphabet<='z')
       return 1;

   if(alphabet>='A' && alphabet<='Z')
       return 1;
       return 0;
}

// palindrome test function

int testPalindrome(char str[],int i,int j){
    
    if(i>j)
       return 1;

    if(alpha(str[i])==0){return testPalindrome(str,i+1,j);}
    if(alpha(str[j])==0){return testPalindrome(str,i,j-1);}
    if(str[i]==str[j])return testPalindrome(str,i+1,j-1);
       return 0;
}

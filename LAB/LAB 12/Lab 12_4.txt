#include <stdio.h>
#include <string.h>
#define SIZE 200

int main(){
  
  char userInput[SIZE];
  int i, counter=0, if_emptyChecker=0;
  char *str, *p;
  char delimiter[] =" \n"; //to delimit space or new line

   for(i=0; i<4;i++){
           fgets(userInput,SIZE,stdin);

           p=userInput;

     while ((str=strtok(p,delimiter))!=NULL){
           if_emptyChecker=1;
                   counter++;
           p = NULL;
  }
          if(if_emptyChecker==0){
          break;
  }
}
printf("%d\n",counter);

return 0;
}

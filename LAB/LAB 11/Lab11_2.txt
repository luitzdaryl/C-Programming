void swapped (int *a, int *b){
    int temp = *a;
    *a = *b;
    *b = temp;
}

void shuffleCards(int card[]){

    srand (time(NULL)); //random seed


    for (int i = 52-1; i > 0; i--){

        int k;
        k = rand() % (i+1);

        swapped(&card[i], &card[k]);
    }
}

int main()
{
    int i;

   for(i=2;i<=1000;i++)
    if(PrimeNumber(2,i)==0)
        printf("%d\n",i);

int PrimeNumber(int i,int num)
{

    if(num==i)
        return 0;
    else
        if(num%i==0)
            return 1;
    else{
        return PrimeNumber(i+1,num);
    }
}

#include<stdio.h> 
# include <stdbool.h>

bool canBalance(int arr1[],int n) 
{
  int l=n;
  for(int i = 0; i < l; i++) 
  {
      int rhs = 0, lhs = 0;
      for(int k = 0; k < l; k++)
      {
          if(k > i)
          {
              lhs += arr1[k];
              }
              else 
              {
                  rhs += arr1[k];
                }
        }
                  
    if(rhs == lhs) 
	{
      return true;
    }
  }
  return false;
}
